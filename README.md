## Install

`npm install --save is-palindrome-string`


## Usage
```js

const isPalindrome = require('is-palindrome-string');

isPalindrome('12321');
//=> true

isPalindrome('fox');
//=> false

isPalindrome('madam');
//=> true

isPalindrome();
//=> TypeError: expects a string got undefined

```
