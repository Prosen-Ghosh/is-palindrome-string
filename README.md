## Install

`npm install --save is-palindrome-string`


## Usage
```js

const palindrome = require('is-palindrome-string');

palindrome('12321')
//=> true

palindrome('fox')
//=> false

palindrome('madam')
//=> true

palindrome()
//=> TypeError: expects a string got undefined

```
